    多平台交叉编译 #

    windows #

    window平台下编译其他平台的程序文件，修改 go env 配置

    # go env
SET CGO_ENABLED=0
SET GOOS=darwin
SET GOARCH=amd64

    # go env
SET CGO_ENABLED=0
SET GOOS=linux
SET GOARCH=amd64

    示例：go env -w CGO_ENABLED=0 GOOS=linux GOARCH=amd64
